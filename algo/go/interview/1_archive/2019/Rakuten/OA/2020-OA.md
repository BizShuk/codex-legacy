# Rakuten 2020 OA

有A B C三個字母 組成不能三個以上連續相同字母之最長字串 


    class Solution {
        private static String a = "a";
        private static String b = "b";
        private static String c = "c";
        private static int maxRepCount = 2;
        
        public String solution(int A, int B, int C) {
            return solution("", null, 0, A,B,C);
        }
        
        public String solution(String current, String lastChar, int rep, int A, int B, int C) {
            String tmpResult = "", finalResult = current;
            
            if (!overRep(a, A, lastChar, rep)) {
                tmpResult = solution(current + a, a, addOneCountIfSameChar(a,lastChar, rep), A-1,B,C);
                if (tmpResult.length() > finalResult.length()) finalResult = tmpResult;
            }
                
            if (!overRep(b, B, lastChar, rep)) {
                tmpResult = solution(current + b, b, addOneCountIfSameChar(b,lastChar, rep), A,B-1,C);
                if (tmpResult.length() > finalResult.length()) finalResult = tmpResult;
            }
                
            if (!overRep(c, C, lastChar, rep)) {
                tmpResult = solution(current + c, c, addOneCountIfSameChar(c,lastChar, rep), A,B,C-1);
                if (tmpResult.length() > finalResult.length()) finalResult = tmpResult;
            }
            
            return finalResult;
        }
        private boolean overRep(String repChar, int remainCount, String lastChar, int rep) {
            if (remainCount <= 0) return true;
            if (repChar.equals(lastChar) && rep >= maxRepCount) return true;
            return false;
        }
        private int addOneCountIfSameChar (String charT, String lastChar, int rep) {
            if (charT.equals(lastChar)) return rep+1;
            return 1;
        }
    }
